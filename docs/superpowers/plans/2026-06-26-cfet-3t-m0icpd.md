### Task 5: Power-row routing ban

**Files:**
- Modify: `src/cellgen/archit/CFET/main.py`

- [ ] **Step 1: Add helper** (near `_ban_other_nets_on_pwr_columns`)

```python
def _ban_signal_on_power_rows(self):
    """M0ICPD: fine row 0 = VSS only, row 5 = VDD only; no signal nets."""
    if self.c_tech.power_config != "M0ICPD":
        return
    self.opt.log_comment("M0ICPD: ban signal nets on power rows 0 and 5")
    m0_idx = self.lgg.layer_to_idx["M0"]
    pwr_rows = self.power_row_indices  # {0: "VSS", 5: "VDD"}
    for net in self.circuit.get_nets(with_power_ground=False):
        for ri in pwr_rows:
            row_coord = self.lgg.row_in_layer("M0", ri)
            for ci in self.lgg.col_indices_in_layer("M0"):
                col_coord = self.lgg.col_in_layer("M0", ci)
                node = (m0_idx, row_coord, col_coord)
                if node in self.node_vars.get(net.name, {}):
                    self.opt.Add(self.node_vars[net.name][node] == 0)
                # also ban edges/arcs touching power rows — iterate self.edge_vars
                for ek, ev in self.edge_vars.items():
                    u, v = ek
                    if u[0] == m0_idx and u[1] == row_coord:
                        self.opt.Add(ev == 0).OnlyEnforceIf(
                            self.net_has_routing.get(net.name, self.opt.NewConstant(1))
                        )  # simplify: unconditional ban for signal nets
```

**Simpler v1 implementation** (preferred if node_vars shape is nested):

```python
def _ban_signal_on_power_rows(self):
    if self.c_tech.power_config != "M0ICPD":
        return
    self.opt.log_comment("M0ICPD: ban signal nets on power rows")
    m0_z = self.lgg.layer_to_idx["M0"]
    for net in self.circuit.get_nets(with_power_ground=False):
        for ri in self.power_row_indices:
            row = self.lgg.row_in_layer("M0", ri)
            for u, v in self.lgg.edges():
                for node in (u, v):
                    if node[0] == m0_z and node[1] == row:
                        ek = (u, v) if (u, v) in self.edge_vars else (v, u)
                        if ek in self.edge_vars:
                            self.opt.Add(self.edge_vars[ek] == 0)
```

Draw full-width M0 boxes at row 0 and row 5 y-coordinates (half-band thickness `m0_pitch / 2`):

```python
def __m0_icpd__(self):
    pitch = self.tech_data.m0_pitch
    half = pitch / 2
    width = self.tech_data.cp_pitch * self.tech_data.col
    # VSS — bottom 0.5T (fine row 0)
    ly_vss = 0.0
    box = Box(0, ly_vss, width, ly_vss + half).pyabox_obj()
    self.cell.shapes(self.M0_layer_idx).insert(box)
    # VDD — top 0.5T (fine row 5)
    ly_vdd = 5 * half  # rows 0..5 span 3*pitch total
    box = Box(0, ly_vdd, width, ly_vdd + half).pyabox_obj()
    self.cell.shapes(self.M0_layer_idx).insert(box)
```

Adjust y formulas to match `draw_power_rail` coordinate system used by existing `__m0_bpr__` (read `self.height` / row layout first).
